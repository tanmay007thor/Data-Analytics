def plot_simulation(in_circle_points, out_circle_points, r):
  plt.figure()

  rectangle = plt.Rectangle((-r, -r), 2*r, 2*r, fc='none', ec='black')
  plt.gca().add_patch(rectangle)

  circle = plt.Circle((0, 0), r, fc='none', ec='black')
  plt.gca().add_patch(circle)

  plt.scatter(in_circle_points[:,0], in_circle_points[:, 1], c='blue', s=0.5, edgecolor=None)
  plt.scatter(out_circle_points[:,0], out_circle_points[:, 1], c='red', s=0.5, edgecolor=None)

  plt.show()

def plot_values(pi_values, n):
  plt.axhline(math.pi, color='green', label='pi value')
  plt.plot(pi_values, color='orange', label='Estimated pi value')

  plt.title(f"{n} Iterations")
  plt.xlabel("Iterations")
  plt.ylabel("Estimated pi value")

  plt.legend()
  plt.show()

def monte_carlo(r, n):
  in_circle = 0
  out_circle = 0

  pi_values = []
  in_circle_points = []
  out_circle_points = []
  for i in range(n):
    for i in range(10):
      x = random.randrange(-r, r)
      y = random.randrange(-r, r)

      if(x**2 + y**2 > r**2):
        out_circle += 1
        out_circle_points.append([x,y])

      else:
        in_circle += 1
        in_circle_points.append([x,y])

    temp = 4*in_circle/(in_circle + out_circle)
    pi_values.append(temp)

  in_circle_points = np.array(in_circle_points)
  out_circle_points = np.array(out_circle_points)

  return pi_values, in_circle_points, out_circle_points
